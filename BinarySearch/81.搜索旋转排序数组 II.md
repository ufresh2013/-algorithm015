# 81.搜索旋转排序数组 II
```js
var search = function(nums, target) {
  let l = 0; r = nums.length - 1;
  while (l <= r) {
    let mid = l + r >> 1
    if (nums[mid] === target) return true
    if (nums[l] <= nums[mid]) {
      if (target >=nums[l] && target < nums[mid]) {
        r = mid - 1
      } else {
        l = mid + 1
      }
    } else {
      if (target > nums[mid] && target <= nums[r]) {
        l = mid + 1
      } else {
        r = mid - 1
      }
    }
  }
  return false
};
```
