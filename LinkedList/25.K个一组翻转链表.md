### K 个一组翻转链表
```js
const rotateRight = (head, k) => {
  if (!head || !head.next || k === 0) return head
  
  // 1. 形成环形链表
  let length = 1
  let l = head
  while (l.next) {
    l = l.next
    length++
  }
  l.next = head // 此时l指向尾节点

  // 2. 尾节点移动 length - k 步
  let step = length - (k % length)
  while (step--) {
    l = l.next
  }
  head = l.next
  l.next = null
  return head
}

const reverseKGroup = (head, k) => {
  let n = k
  while (n--) {
    rotateRight = (head, k)
  }
  return head
}
```
