# 合并两个有序链表
```js
var mergeTwoLists = function(l1, l2) {
  let curNode = new ListNode('head')
  let l3 = curNode

  while (l1 !== null || l2 !== null) {
    if (l1 === null) {
      curNode.next = l2
      break
    }

    if (l2 === null) {
      curNode.next = l1
      break
    }

    if (l1.val > l2.val) {
      curNode.next = l2
      l2 = l2.next
    } else {
      curNode.next = l1
      l1 = l1.next
    }
    curNode = curNode.next
  }
  return l3.next
};
```
